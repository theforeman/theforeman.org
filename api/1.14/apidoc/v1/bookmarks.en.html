      "id": 980190962,
      "owner_id": null,
      "owner_type": null

  }

POST /api/bookmarks
Create a bookmark.

Examples

POST /api/bookmarks
{
  "bookmark": {
    "public": false,
    "controller": "hosts",
    "name": "foo-bar",
    "query": "bar"
  }
}
201
{
  "bookmark": {
    "name": "foo-bar",
    "controller": "hosts",
    "query": "bar",
    "public": false,
    "id": 980190963,
    "owner_id": 886836129,
    "owner_type": "User"
  }
}
