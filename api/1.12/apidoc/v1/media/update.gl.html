PUT /api/media/:id
Update a medium.

Examples

PUT /api/media/980190962
{
  "name": "CentOS 5.4",
  "medium": {
    "name": "CentOS 5.4"
  }
}
200
{
  "medium": {
    "id": 980190962,
    "name": "CentOS 5.4",
    "path": "http://mirror.centos.org/centos/6.0/os/$arch",
    "created_at": "2016-06-15T12:53:26.000Z",
    "updated_at": "2016-06-15T12:53:26.000Z",
    "media_path": null,
    "config_path": null,
    "image_path": null,
    "os_family": null
  }
}
