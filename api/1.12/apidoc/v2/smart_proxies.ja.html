POST /api/smart_proxies/:id/import_puppetclasses

Examples

}

    {
      "name": "env1",

        "updated"
      ],
      "updated_puppetclasses": [
        "a"
      ]
    },
    {
      "name": "env2",
      "actions": [
        "updated"
      ],
      "updated_puppetclasses": [
        "a"
      ]
    }
  ]
}

Required
