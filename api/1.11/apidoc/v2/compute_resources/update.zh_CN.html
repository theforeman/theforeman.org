PUT /api/compute_resources/:id
Update a compute resource

    "description": "new_description"
  }
}

  "url": "test:///default",

  "display_type": "vnc",
  "set_console_password": true,

  "compute_attributes": [],
  "locations": [],
  "organizations": []
}
