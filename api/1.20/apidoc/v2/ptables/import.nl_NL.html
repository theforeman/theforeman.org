POST /api/ptables/import
Import a provisioning template

Examples

POST /api/ptables/import
{
  "ptable": {
    "name": "ptable148",
    "template": "b"
  }
}
201
{
  "template": "b",
  "name": "ptable148",
  "id": 1007981804,
  "snippet": false,
  "template_kind_id": null,

  "updated_at": "2018-11-15T19:02:39.656Z",
  "locked": false,
  "default": false,
  "vendor": null,
  "os_family": "Redhat"
}
