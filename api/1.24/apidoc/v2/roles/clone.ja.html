POST /api/roles/:id/clone
ロールのクローン作成

例

POST /api/roles/16/clone
{
  "name": "New Role",
  "role": {
    "description": "updated description",
    "organization_ids": [
      61304186
    ],
    "name": "New Role"
  }
}
201
{
  "id": 17,
  "name": "New Role",
  "builtin": 0,
  "description": "updated description",
  "origin": null,
  "cloned_from_id": 16
}
