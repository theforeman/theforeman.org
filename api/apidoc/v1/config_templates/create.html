POST /api/config_templates
Create a template

Examples

POST /api/config_templates
{
  "config_template": {
    "template": "This is a test template",
    "template_kind_id": "698706415",
    "name": "RandomName"
  }
}
200
{
  "config_template": {
    "name": "RandomName",
    "template": "This is a test template",
    "snippet": null,
    "audit_comment": null,
    "id": 1007981702,
    "template_kind": {
      "name": "iPXE",
      "id": 698706415
    }
  }
}
