Examples

}

        "new"

        "a"

Params

Param name Description
id
required

optional

environment_id
optional
Value: Must be String

optional

optional

    "url": "http://server:8443"

}

{
  "created_at": "2014-08-07T20:41:40Z",
  "id": 982965215,

  "updated_at": "2014-08-07T20:41:40Z",

}
