POST /api/compute_resources
Create a compute resource.

Examples

POST /api/compute_resources
{
  "compute_resource": {
    "name": "special_compute",
    "region": "eu-west-1",
    "user": "user@example.com",
    "password": "secret"
  }
}
200
{
  "id": 980190964,
  "name": "special_compute",
  "provider": "EC2",
  "description": null,
  "url": "eu-west-1",
  "created_at": "2014-07-23T13:31:25Z",
  "updated_at": "2014-07-23T13:31:25Z",
  "access_key": "user@example.com",
  "region": "eu-west-1",
  "locations": [],
  "organizations": [],
  "images": [],
  "compute_attributes": []
}
