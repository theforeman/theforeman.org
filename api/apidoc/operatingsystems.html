POST /api/operatingsystems
Create an OS.

Examples

POST /api/operatingsystems
{
  "operatingsystem": {
    "minor": "2",
    "name": "awsome_os",
    "major": "1"
  }
}
200
{
  "operatingsystem": {
    "name": "awsome_os",
    "minor": "2",
    "id": 1073012829,
    "family": null,
    "ptables": [],
    "os_default_templates": [],
    "release_name": null,
    "config_templates": [],
    "architectures": [],
    "major": "1",
    "media": []
  }
}
