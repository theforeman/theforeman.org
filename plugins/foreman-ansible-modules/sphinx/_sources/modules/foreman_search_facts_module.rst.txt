Examples
--------

.. code-block:: yaml+jinja

    
    - name: "Read a Setting"
      foreman_search_facts:
        username: "admin"
        password: "changeme"
        server_url: "https://foreman.example.com"
        resource: Setting
        search: name = http_proxy
      register: result
    - debug:
        var: result.response[0].value

    - name: "Read all Registries"
      foreman_search_facts:
        username: "admin"
        password: "changeme"
        server_url: "https://foreman.example.com"
        resource: Registry
      register: result
    - debug:
        var: item.name
      with_items: result.response
